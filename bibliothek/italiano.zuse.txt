# ZUSE BIBLIOTECA STANDARD (ITALIANO)

USARE tkinter COME _tk
USARE turtle COME _turtle
USARE random COME _rnd
USARE time COME _zeit

CLASSE Finestra:
    DEFINIRE CREARE(titolo, larghezza, altezza):
        MIO.obj = _tk.Tk()
        MIO.obj.title(titolo)
        MIO.obj.geometry(str(larghezza) + "x" + str(altezza))
    FINE FUNZIONE

    DEFINIRE avvia():
        MIO.obj.mainloop()
    FINE FUNZIONE
FINE CLASSE

CLASSE Pittore:
    DEFINIRE CREARE():
        MIO.t = _turtle.Turtle()
        MIO.t.shape("turtle")
        MIO.t.speed(3)
        MIO.screen = _turtle.Screen()
        MIO.screen.title("Scuola d'Arte Zuse")
    FINE FUNZIONE

    DEFINIRE muovi(passi):
        MIO.t.forward(passi)
    FINE FUNZIONE

    DEFINIRE indietro(passi):
        MIO.t.backward(passi)
    FINE FUNZIONE

    DEFINIRE gira_sinistra(gradi):
        MIO.t.left(gradi)
    FINE FUNZIONE

    DEFINIRE gira_destra(gradi):
        MIO.t.right(gradi)
    FINE FUNZIONE

    DEFINIRE alza_penna():
        MIO.t.penup()
    FINE FUNZIONE

    DEFINIRE abbassa_penna():
        MIO.t.pendown()
    FINE FUNZIONE

    DEFINIRE colore(c):
        SE c == "rosso" ALLORA
            MIO.t.color("red")
        ALTRIMENTI SE c == "blu" ALLORA
            MIO.t.color("blue")
        ALTRIMENTI SE c == "verde" ALLORA
            MIO.t.color("green")
        ALTRIMENTI SE c == "giallo" ALLORA
            MIO.t.color("yellow")
        ALTRIMENTI
            MIO.t.color(c)
        FINE SE
    FINE FUNZIONE
    
    DEFINIRE spessore(s):
        MIO.t.pensize(s)
    FINE FUNZIONE

    DEFINIRE cerchio(raggio):
        MIO.t.circle(raggio)
    FINE FUNZIONE
    
    DEFINIRE fatto():
        _turtle.done()
    FINE FUNZIONE
FINE CLASSE

DEFINIRE aspettare(secondi):
    _zeit.sleep(secondi)
FINE FUNZIONE
